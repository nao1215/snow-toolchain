    integer literal *

    boolean literal *

    *)

    (*

    identifier *

    *)

    (*

    'if' keyword *

    *)

    (*

    'else' keyword *

    *)

    (*

    'while' keyword *

    *)

    (*

    'return' keyword *

Token type representing the atomic elements of Bosse source.

val is_digit : char -> bool

Check if character is a digit (0-9).

val is_alpha : char -> bool

Check if character is alphabetic or underscore.

val keywords : (string * token) list

List of reserved keywords and boolean literals.

val skip : string -> int -> int -> int

Skip whitespace and comments starting from index i; returns next non-whitespace index.

val lex : string -> token list

Perform lexical analysis on input and return the list of tokens.
