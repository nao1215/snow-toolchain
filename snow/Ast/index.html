type typ =
  1. | TInt
    (*

    32-bit signed integer

    *)
  2. | TBool
    (*

    Boolean type

    *)
  3. | TArray of typ * int option
    (*

    Array of typ, fixed-length Some n or variable-length None

    *)

type expr =
  1. | EInt of int
    (*

    Integer literal

    *)
  2. | EBool of bool
    (*

    Boolean literal

    *)
  3. | EVar of string
    (*

    Variable reference

    *)
  4. | EArrayAccess of expr * expr
    (*

    Array indexing aexpr

    *)
  5. | ECall of string * expr list
    (*

    Function call name(args...)

    *)
  6. | EBinOp of binop * expr * expr
    (*

    Binary operation

    *)
  7. | EUnOp of unop * expr
    (*

    Unary operation

    *)

Expressions in Bosse.

type stmt =
  1. | SVarDecl of string * typ * expr option
    (*

    Variable declaration: name, type, optional initializer

    *)
  2. | SAssign of expr * expr
    (*

    Assignment: lhs = rhs

    *)
  3. | SExpr of expr
    (*

    Expression statement

    *)
  4. | SIf of expr * stmt list * stmt list option
    (*

    If statement: condition, then-branch, optional else-branch

    *)
  5. | SWhile of expr * stmt list
    (*

    While loop: condition and body

    *)
  6. | SReturn of expr
    (*

    Return statement

    *)

Statements in Bosse.
